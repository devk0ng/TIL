#include<bits/stdc++.h>

using namespace std;

int N,M;
vector<int> true_person;
vector<int> result;

int BFS(vector< set<int> > &g) {
	int special_man=0;
	
	vector<int> visited(N+1, 0);
	
	for(int i=0 ; i<true_person.size() ; i++) {
		queue<int> q;
		
		q.push(true_person[i]);

		while(!q.empty()) {
			int start = q.front();
			
			q.pop();
			
			visited[start] = 1;
			result.push_back(start);
			special_man++;
			
			for(auto iter = g[start].begin(); iter != g[start].end(); iter++){
        		int next = *iter;

				if(visited[next] == 1)
					continue;
				
				q.push(next);
				result.push_back(next);
				visited[next] = 1;
				special_man++;
		    }


		}	
	}
	
	return special_man;
}

void print(vector< set<int> > &g) {
	for(int i=1 ; i<=N ; i++) {
		cout<<i<<" : ";
		
		for(auto iter=g[i].begin() ; iter!=g[i].end() ; iter++) {
			cout<<*iter<<" ";
		}
		cout<<"\n";
	}
	
}
